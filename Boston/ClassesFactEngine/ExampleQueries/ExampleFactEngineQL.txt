       Room.Number,
       Position.Name,
       School.Name
FROM Lecturer,
     Room,
     Position,
     School,
     Faculty
WHERE Lecturer.RoomId = Room.RoomId
  AND Lecturer.PositionId = Position.Id
  AND Lecturer.SchoolId = School.SchoolId
  AND School.FacultyId = Faculty.FacultyId
  AND Faculty.Name = 'IT'
  AND Lecturer.FacultyId = Faculty.FacultyId
